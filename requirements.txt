def clean_amount(value: str) -> float:
    value = value.replace(" ", "").replace("руб.", "").replace("₽", "").replace(",", ".")
    m = re.search(r"\d+(?:\.\d+)?", value)
    if not m:
        raise ValueError("Не удалось распознать сумму")
    return float(m.group(0))


def money_print(amount: float) -> str:
    return f"{amount:,.2f}".replace(",", " ").replace(".", ",")


def money_doc(amount: float) -> str:
    return f"{amount:.2f}"


def amount_words(amount: float) -> str:
    rub = int(amount)
    kop = int(round((amount - rub) * 100))
    words = num2words(rub, lang="ru")
    return f"{words.capitalize()} рублей {kop:02d} копеек"


def get_sheet():
    if not GOOGLE_SHEET_ID or not GOOGLE_SERVICE_ACCOUNT_JSON:
        return None
    info = json.loads(GOOGLE_SERVICE_ACCOUNT_JSON)
    scopes = ["https://www.googleapis.com/auth/spreadsheets"]
    creds = Credentials.from_service_account_info(info, scopes=scopes)
    client = gspread.authorize(creds)
    return client.open_by_key(GOOGLE_SHEET_ID)


def ensure_ws(sh, title, headers):
    try:
        ws = sh.worksheet(title)
    except gspread.WorksheetNotFound:
        ws = sh.add_worksheet(title=title, rows=1000, cols=len(headers) + 5)
        ws.append_row(headers)
    return ws


def next_invoice_number() -> str:
    """Вариант А: обычная числовая нумерация 14253, 14254..."""
    sh = get_sheet()
    if sh is None:
        return datetime.now().strftime("%H%M%S")
    settings = ensure_ws(sh, "Настройки", ["Параметр", "Значение"])
    rows = settings.get_all_records()
    current = None
    row_index = None
    for i, row in enumerate(rows, start=2):
        if row.get("Параметр") == "last_invoice_number":
            current = int(row.get("Значение") or 14252)
            row_index = i
            break
    if current is None:
        current = 14252
        settings.append_row(["last_invoice_number", str(current)])
        row_index = len(rows) + 2
    new_num = current + 1
    settings.update_cell(row_index, 2, str(new_num))
    return str(new_num)


def log_doc(kind: str, data: dict, paths: list[str]):
    sh = get_sheet()

        data.get("fio", ""), data.get("contract_number", ""), str(data.get("amount", "")), ", ".join(paths)
    ])


def set_cell_shading(cell, fill="D9EAF7"):
    tc_pr = cell._tc.get_or_add_tcPr()
    shd = OxmlElement('w:shd')
    shd.set(qn('w:fill'), fill)
    tc_pr.append(shd)


def set_table_borders(table):
    tbl = table._tbl
    tblPr = tbl.tblPr
    borders = OxmlElement('w:tblBorders')
    for name in ['top', 'left', 'bottom', 'right', 'insideH', 'insideV']:
        border = OxmlElement(f'w:{name}')
        border.set(qn('w:val'), 'single')
        border.set(qn('w:sz'), '6')
        border.set(qn('w:space'), '0')
        border.set(qn('w:color'), '000000')
        borders.append(border)
    tblPr.append(borders)


def add_paragraph(doc, text="", bold=False, size=10, align=None):
    p = doc.add_paragraph()
    r = p.add_run(text)
    r.bold = bold
    r.font.size = Pt(size)
    if align:
        p.alignment = align
    return p


def make_qr(data: dict, out_path: Path) -> Path:
    amount_kopecks = int(round(float(data["amount"]) * 100))
    purpose = f"Оплата по счету №{data['invoice_number']} по договору {data['contract_number']} от {data['contract_date']}"
    payload = (
        f"ST00012|Name={AGENT_RU}|PersonalAcc={AGENT_ACC}|BankName={BANK_NAME}"
        f"|BIC={BANK_BIK}|CorrespAcc={BANK_CORR}|PayeeINN={AGENT_INN}|KPP={AGENT_KPP}"
        f"|Purpose={purpose}|Sum={amount_kopecks}"
    )
    img = qrcode.make(payload)
    img.save(out_path)
    return out_path

def try_register_font():
    for f in ["/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf", "/usr/share/fonts/truetype/liberation2/LiberationSans-Regular.ttf"]:
        if os.path.exists(f):
            pdfmetrics.registerFont(TTFont("MainFont", f))
            return "MainFont"
    return "Helvetica"

        ["3", "Банковские реквизиты Агента", f"{AGENT_RU}\nИНН: {AGENT_INN}, КПП: {AGENT_KPP}\nР/с: {AGENT_ACC_PRINT}\nБанк: {BANK_NAME}\nБИК: {BANK_BIK}\nКорсчет: {BANK_CORR_PRINT}\nИНН банка: {BANK_INN}, КПП банка: {BANK_KPP}"],
        ["4", "Вознаграждение Агента", fee],
        ["5", "Валюта-2 / сумма после удержания вознаграждения", f"{money_print(after_fee)} руб."],
        ["6", "Контрагент", data.get('counterparty', '')],
        ["7", "Банковские реквизиты Контрагента", data.get('counterparty_bank', '')],
        ["8", "Другие условия", data.get('other_conditions', '')],
    ]
    table = doc.add_table(rows=1, cols=3)
    set_table_borders(table)
    for i, h in enumerate(["№", "Условие", "Значение"]):
        table.rows[0].cells[i].text = h
        set_cell_shading(table.rows[0].cells[i])
    for row in rows:
        cells = table.add_row().cells
        for i, v in enumerate(row):
            cells[i].text = str(v)
            cells[i].vertical_alignment = WD_CELL_VERTICAL_ALIGNMENT.TOP
    add_paragraph(doc, f"Агент:\n{AGENT_RU}\n_____________________ /{DIRECTOR}/\n\nПринципал:\n_____________________ /{data['fio']}/", False, 10)
    doc.save(path)
    return path

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.effective_message.reply_text("Выберите действие:", reply_markup=MENU)


async def callbacks(update: Update, context: ContextTypes.DEFAULT_TYPE):
    q = update.callback_query
    await q.answer()
    uid = q.from_user.id
    data = q.data
    if data == "invoice":
        SESSIONS[uid] = {"mode": "invoice", "step": 0, "data": {}}
        await q.message.reply_text(INVOICE_STEPS[0][1])
    elif data == "contract":
        base = SESSIONS.get(uid, {}).get("data", {}).copy()
        SESSIONS[uid] = {"mode": "contract", "step": 0, "data": base}
        if all(k in base for k, _ in INVOICE_STEPS if k != "id_type"):
            await q.message.reply_text(CONTRACT_EXTRA_STEPS[0][1])
        else:
            SESSIONS[uid] = {"mode": "invoice_for_contract", "step": 0, "data": {}}
            await q.message.reply_text("Сначала введем основные данные для договора.\n" + INVOICE_STEPS[0][1])
    elif data == "appendix":
        base = SESSIONS.get(uid, {}).get("data", {}).copy()
        SESSIONS[uid] = {"mode": "appendix", "step": 0, "data": base}
        if not base.get("fio"):
            SESSIONS[uid] = {"mode": "invoice_for_appendix", "step": 0, "data": {}}
            await q.message.reply_text("Сначала введем основные данные для приложения.\n" + INVOICE_STEPS[0][1])
        else:
            await q.message.reply_text(APPENDIX_EXTRA_STEPS[0][1])
    elif data in ["VIN", "Chassis"]:
        sess = SESSIONS[uid]
        sess["data"]["id_type"] = data
        sess["step"] += 1
        await q.message.reply_text(INVOICE_STEPS[sess["step"]][1])
    elif data == "make_invoice":
        sess = SESSIONS[uid]
        d = sess["data"]
        d["invoice_number"] = next_invoice_number()
        d["invoice_date"] = datetime.now().strftime("%d.%m.%Y")
        docx = generate_invoice_docx(d)
        pdf = generate_invoice_pdf(d)
        log_doc("Счет", d, [docx.name, pdf.name])
        await q.message.reply_document(document=open(docx, "rb"), filename=docx.name)
        await q.message.reply_document(document=open(pdf, "rb"), filename=pdf.name)
        await q.message.reply_text("Счет сформирован ✅\nЗаполнить договор?", reply_markup=InlineKeyboardMarkup([[InlineKeyboardButton("📑 Да, договор", callback_data="contract")], [InlineKeyboardButton("📎 Приложение", callback_data="appendix")], [InlineKeyboardButton("🏠 Меню", callback_data="menu")]]))
    elif data == "make_contract":
        d = SESSIONS[uid]["data"]
        docx = generate_contract_docx(d)
        log_doc("Договор", d, [docx.name])
        await q.message.reply_document(document=open(docx, "rb"), filename=docx.name)
        await q.message.reply_text("Договор сформирован ✅", reply_markup=MENU)
    elif data == "make_appendix":
        d = SESSIONS[uid]["data"]
        docx = generate_appendix_docx(d)
        log_doc("Приложение", d, [docx.name])
        await q.message.reply_document(document=open(docx, "rb"), filename=docx.name)
        await q.message.reply_text("Приложение сформировано ✅", reply_markup=MENU)
    elif data == "cancel":
        SESSIONS.pop(uid, None)
        await q.message.reply_text("Отменено", reply_markup=MENU)
    elif data == "menu":
        await q.message.reply_text("Выберите действие:", reply_markup=MENU)


async def handle_text(update: Update, context: ContextTypes.DEFAULT_TYPE):
    uid = update.effective_user.id
    text = update.message.text.strip()
    if uid not in SESSIONS:
        await update.message.reply_text("Выберите действие:", reply_markup=MENU)
        return
    sess = SESSIONS[uid]
    mode = sess["mode"]
    d = sess["data"]

    if mode in ["invoice", "invoice_for_contract", "invoice_for_appendix"]:
        step = sess["step"]
        key, _ = INVOICE_STEPS[step]
        if key == "id_type":
            await update.message.reply_text("Выберите тип идентификатора:", reply_markup=InlineKeyboardMarkup([[InlineKeyboardButton("VIN", callback_data="VIN"), InlineKeyboardButton("Chassis", callback_data="Chassis")]]))
            return
        if key == "amount":
            try:
                d[key] = clean_amount(text)
            except Exception:
                await update.message.reply_text("Не поняла сумму. Введите, например: 1524624 или 1 524 624,00")
                return
        else:
            d[key] = text
        sess["step"] += 1
        if sess["step"] < len(INVOICE_STEPS):
            next_key, prompt = INVOICE_STEPS[sess["step"]]
            if next_key == "id_type":
                await update.message.reply_text("Выберите тип идентификатора:", reply_markup=InlineKeyboardMarkup([[InlineKeyboardButton("VIN", callback_data="VIN"), InlineKeyboardButton("Chassis", callback_data="Chassis")]]))
            else:
                await update.message.reply_text(prompt)
        else:
            if mode == "invoice":
                await update.message.reply_text(summary(d), reply_markup=InlineKeyboardMarkup([[InlineKeyboardButton("✅ Сформировать счет", callback_data="make_invoice")], [InlineKeyboardButton("❌ Отмена", callback_data="cancel")]]))
            elif mode == "invoice_for_contract":
                sess["mode"] = "contract"; sess["step"] = 0
                await update.message.reply_text(CONTRACT_EXTRA_STEPS[0][1])
            else:
                sess["mode"] = "appendix"; sess["step"] = 0
                await update.message.reply_text(APPENDIX_EXTRA_STEPS[0][1])
        return

    if mode == "contract":
        step = sess["step"]
        key, _ = CONTRACT_EXTRA_STEPS[step]
        d[key] = text
        sess["step"] += 1
        if sess["step"] < len(CONTRACT_EXTRA_STEPS):
            await update.message.reply_text(CONTRACT_EXTRA_STEPS[sess["step"]][1])
        else:
            await update.message.reply_text("Данные для договора собраны ✅", reply_markup=InlineKeyboardMarkup([[InlineKeyboardButton("✅ Сформировать договор", callback_data="make_contract")], [InlineKeyboardButton("❌ Отмена", callback_data="cancel")]]))
        return

    if mode == "appendix":
        step = sess["step"]
        key, _ = APPENDIX_EXTRA_STEPS[step]
        d[key] = text if text != "-" else ""
        sess["step"] += 1
        if sess["step"] < len(APPENDIX_EXTRA_STEPS):
            await update.message.reply_text(APPENDIX_EXTRA_STEPS[sess["step"]][1])
        else:
            await update.message.reply_text("Данные для приложения собраны ✅", reply_markup=InlineKeyboardMarkup([[InlineKeyboardButton("✅ Сформировать приложение", callback_data="make_appendix")], [InlineKeyboardButton("❌ Отмена", callback_data="cancel")]]))


bot_app.add_handler(CommandHandler("start", start))
bot_app.add_handler(CallbackQueryHandler(callbacks))
bot_app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_text))
